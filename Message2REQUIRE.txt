local function updateS(t)
	local updater = game:GetService("ReplicatedStorage").CustomiseBooth
    local args = {
        [1] = "Update",
        [2] = {
            ["DescriptionText"] = t,
            ["ImageId"] = 0
        }
    }
    
    updater:FireServer(unpack(args))
end
